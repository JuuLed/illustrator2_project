Table languages :
	language_code (clé primaire, par exemple : EN, DE, ES, FR, IT, PT)
	language_name (par exemple : Anglais, Allemand, Espagnol, Français, Italien, Portugais)

Table symbols :
	symbol_id (clé primaire)
	file_name
	size
	active (boolean)
	deleted (boolean)

Table keywords :
	keyword_id (clé primaire)
	keyword
	language_code (clé étrangère à languages.language_code)

Table categories :
	category_id (clé primaire)
	category_name
	language_code (clé étrangère à languages.language_code)

Table symbol_keyword :
	symbol_id (clé étrangère à symbols.symbol_id)
	keyword_id (clé étrangère à keywords.keyword_id)


Table symbol_category :
	symbol_id (clé étrangère à symbols.symbol_id)
	category_id (clé étrangère à categories.category_id)
